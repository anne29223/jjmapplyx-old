<script>
document.getElementById("syncJobsBtn").addEventListener("click", async (e) => {
  e.preventDefault();
  const status = document.getElementById("syncStatus");
  status.innerText = "Syncing jobs…";

  try {
    const res = await fetch("/dashboard/jobs/sync");
    const data = await res.json();

    if (data.success) {
      status.innerText = `✅ Synced ${data.synced} jobs`;
      setTimeout(() => location.reload(), 1000);
    } else {
      status.innerText = `❌ Error: ${data.error}`;
    }
  } catch (err) {
    status.innerText = `❌ Fetch error: ${err.message}`;
  }
});
</script>
